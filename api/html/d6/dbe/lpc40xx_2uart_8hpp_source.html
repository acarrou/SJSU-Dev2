49 static constexpr auto GenerateOversizedBaudTable() noexcept
50 {
51  constexpr uint32_t numerator_min = 1;
52  constexpr uint32_t denominator_min = 1;
53  constexpr uint32_t denominator_max = 15;
54 
55  TableIntermediate table{};
56 
57  // First element is the fractional 1.0f value. This allows baud rates that
58  // have dividers that are already integers, on the first iteration of the
59  // fractional value, this will match and the loop will stop.
60  table.element[table.valid_entries++] = UartFractionals{
61  .fraction = 1,
62  .byte_code = (1 << 4) | (0),
63  };
64 
65  // Skip zero numerator
66  for (uint32_t denominator = denominator_min; denominator < denominator_max;
67  denominator++)
68  {
69  for (uint32_t numerator = numerator_min; numerator < denominator;
70  numerator++)
71  {
72  float fraction = 1.0f + static_cast<float>(numerator) /
73  static_cast<float>(denominator);
74  if (1.1f <= fraction && fraction <= 1.9f)
75  {
76  table.element[table.valid_entries++] = UartFractionals{
77  .fraction = fraction,
78  .byte_code = static_cast<uint8_t>((denominator << 4) | (numerator)),
79  };
80  }
81  }
82  }
83 
84  return table;
85 }

195 constexpr static UartCalibration_t GenerateUartCalibration(
196  uint32_t baud_rate,
197  units::frequency::hertz_t peripheral_frequency)
198 {
199  uint32_t integer_peripheral_frequency =
200  units::unit_cast<uint32_t>(peripheral_frequency);
201  States state = States::kCalculateIntegerDivideLatch;
202  UartCalibration_t uart_calibration;
203  float baud_rate_float = static_cast<float>(baud_rate);
204  float divide_estimate = 0;
205  float decimal = kBaudTable[0].fraction;
206  size_t table_position = 0;
207  while (state != States::kDone)
208  {
209  switch (state)
210  {
211  case States::kCalculateIntegerDivideLatch:
212  {
213  divide_estimate =
214  DividerEstimate(baud_rate_float, 1, integer_peripheral_frequency);
215 
216  if (divide_estimate < 1.0f)
217  {
218  uart_calibration.divide_latch = 0;
219  state = States::kDone;
220  }
221  else if (IsDecimal(divide_estimate))
222  {
223  uart_calibration.divide_latch =
224  static_cast<uint32_t>(divide_estimate);
225  state = States::kDone;
226  }
227  else
228  {
229  state = States::kCalculateDivideLatchWithDecimal;
230  }
231  break;
232  }
233  case States::kCalculateDivideLatchWithDecimal:
234  {
235  divide_estimate = RoundFloat(DividerEstimate(
236  baud_rate_float, decimal, integer_peripheral_frequency));
237  decimal = FractionalEstimate(
238  baud_rate_float, divide_estimate, integer_peripheral_frequency);
239  if (1.1f <= decimal && decimal <= 1.9f)
240  {
241  state = States::kGenerateFractionFromDecimal;
242  }
243  else
244  {
245  state = States::kDecimalFailedGenerateNewDecimal;
246  }
247  break;
248  }
249  case States::kDecimalFailedGenerateNewDecimal:
250  {
251  if (table_position >= kBaudTable.size())
252  {
253  state = States::kDone;
254  }
255  decimal = kBaudTable[table_position++].fraction;
256  state = States::kCalculateDivideLatchWithDecimal;
257  break;
258  }
259  case States::kGenerateFractionFromDecimal:
260  {
261  uart_calibration = FindClosestFractional(decimal);
262  uart_calibration.divide_latch = static_cast<uint32_t>(divide_estimate);
263  state = States::kDone;
264  break;
265  }
266  case States::kDone:
267  default: break;
268  }
269  }
270  return uart_calibration;
271 }
