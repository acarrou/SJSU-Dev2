137  void Transaction(Transaction_t transaction) override
138  {
139  auto control = bit::Register(&i2c_.registers->CR1);
140  auto status = bit::Register(&i2c_.registers->SR1);
141  [[maybe_unused]] uint32_t status2 = 0;
142 
143  timeout_.SetNewTimeout(transaction.timeout);
144 
145  Start(transaction);
146 
147  // Step 7.a Stream out data to i2c bus if out_length > 0
148  for (size_t i = 0; i < transaction.out_length; i++)
149  {
150  // Step 7.b. Wait for data to finish being sent
151  while (!status.Read(SR1::kTxDataRegisterEmpty))
152  {
153  if (AcknowledgeFailure())
154  {
155  control.Set(CR1::kStop).Save();
156  return;
157  }
158  }
159 
160  i2c_.registers->DR = transaction.data_out[i];
161  }
162 
163  // Check if this transaction is a write-then-read operation
164  if (transaction.repeated && transaction.in_length > 0)
165  {
166  transaction.operation = Operation::kRead;
167  Start(transaction);
168  }
169 
170  if (transaction.in_length > 0)
171  {
172  // Step 8. Read data from bus if in_length > 0
173  for (size_t i = 0; i < transaction.in_length - 1; i++)
174  {
175  transaction.data_in[i] = GetData();
176  }
177 
178  control.Clear(CR1::kAcknowledge).Set(CR1::kStop).Save();
179 
180  transaction.data_in[transaction.in_length - 1] = GetData();
181  }
182  else
183  {
184  control.Set(CR1::kStop).Save();
185  }
186 
187  // Wait until the transaction is complete.
188  return;
189  }

204  void Start(Transaction_t & transaction)
205  {
206  auto control = bit::Register(&i2c_.registers->CR1);
207  auto status = bit::Register(&i2c_.registers->SR1);
208  auto status2 = bit::Register(&i2c_.registers->SR2);
209 
210  [[maybe_unused]] uint32_t read_value = 0;
211 
212  // Step 1. Start the transaction
213  control.Set(CR1::kStart).Set(CR1::kAcknowledge).Save();
214 
215  // Step 2. Wait for start sent
216  while (!status.Read(SR1::kStartSent))
217  {
218  ThrowErrorIfPresent();
219  }
220 
221  // Step X.
222  while (!status2.Read(SR2::kBusy) && !status2.Read(SR2::kIsMaster))
223  {
224  ThrowErrorIfPresent();
225  }
226 
227  // Step 3. Set data register to address
228  i2c_.registers->DR = transaction.GetProperAddress();
229 
230  // Step 4. Wait for address to be sent
231  while (!status.Read(SR1::kAddressSent))
232  {
233  ThrowErrorIfPresent();
234  if (AcknowledgeFailure())
235  {
236  throw CommonErrors::kDeviceNotFound;
237  }
238  }
239 
240  read_value = i2c_.registers->SR2;
241  }
242 
243  bool AcknowledgeFailure()
244  {
245  auto status = bit::Register(&i2c_.registers->SR1);
246  bool failure = status.Read(SR1::kAcknowledgeFailure);
247  if (failure)
248  {
249  status.Clear(SR1::kAcknowledgeFailure).Save();
250  }
251  return failure;
252  }

281  void ConfigureClockRate()
282  {
283  // Calculating and setting the I2C Clock rate
284  // Weight the high side duty cycle more than the lower side by 30% in
285  // order to give more time for the bus to charge up.
286  const auto kPeripheralFrequency =
287  sjsu::SystemController::GetPlatformController().GetClockRate(i2c_.id);
288 
289  if (kPeripheralFrequency <= 4_MHz)
290  {
291  throw I2c::CommonErrors::kClockRateNotPossible;
292  }
293 
294  // Cut clock divider in half for HIGH and LOW time
295  const float kClockDivider = kPeripheralFrequency / (settings.frequency * 3);
296 
297  bit::Register(&i2c_.registers->CCR)
298  .Insert(0, CCR::kDuty)
299  .Insert(1, CCR::kModeSelect)
300  .Insert(static_cast<uint16_t>(kClockDivider), CCR::kClockControl)
301  .Save();
302 
303  i2c_.registers->TRISE =
304  static_cast<uint16_t>(((kPeripheralFrequency * 300) / 1'000_Hz) + 1);
305  }
306 
307  uint8_t GetData()
308  {
309  auto status = bit::Register(&i2c_.registers->SR1);
310  // Step 7.b. Wait for data to finish being sent
311  while (!status.Read(SR1::kRxDataRegisterNotEmpty))
312  {
313  ThrowErrorIfPresent();
314  }
315 
316  return static_cast<uint8_t>(i2c_.registers->DR);
317  }
