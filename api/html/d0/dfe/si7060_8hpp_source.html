68 
69  uint8_t most_significant_register;
70  uint8_t least_significant_register;
71 
72  // The register will enable the device to collect data once
73  // and automatically sets the stop bit to 0 (2nd bit).
74  i2c_.Write(address_, { kOneBurstRegister, 0x04 });
75 
76  // Enable I2C automatic address increment on read (must be done every time)
77  i2c_.Write(address_, { kAutomaticBitRegister, 0x01 });
78 
79  // These need to be in separate transactions
80  i2c_.WriteThenRead(
81  address_, { kMostSignificantRegister }, &most_significant_register, 1);
82  i2c_.Read(address_, &least_significant_register, 1);
83 
84  // The write and read operation sets the most significant bit to one,
85  // telling the register that the data has been read.
86  // Therefore must clear the most significant bit from above.
87  most_significant_register = bit::Clear(most_significant_register, 7);
88 
89  // Combine temperature register data
90  int32_t temperature_data =
91  (most_significant_register << 8) | least_significant_register;
92 
93  // The required computation after bit shifting.
94  // Formula can be found below, see page 4:
95  // datasheets/Temperature-Sensor/si7060-datasheets.pdf
96  float acquired_temperature =
97  static_cast<float>(temperature_data - kSubtractTemperatureData);
98  auto temperature =
99  units::temperature::celsius_t{ (acquired_temperature / 160.0f) +
100  55.0f };
101 
102  return temperature;
103  }
